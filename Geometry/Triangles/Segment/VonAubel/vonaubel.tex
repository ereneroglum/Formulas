\begin{equation}
    \begin{aligned}    
        \frac{a_1}{a_2} + \frac{b_1}{b_2} = \frac{x_1}{x_2} \\
    \end{aligned}
\end{equation}

\ref{fig:vonaubtrig2} Üçgen figüründe de görülebileceği gibi,
\begin{equation}
    \begin{aligned}
        \frac{x_1}{x_1 + x_2} + \frac{y_1}{y_1 + y_2} + \frac{z_1}{z_1 + z_2} = 2 \\
        \frac{x_2}{x_1 + x_2} + \frac{y_2}{y_1 + y_2} + \frac{z_2}{z_1 + z_2} = 1 \\
    \end{aligned}
\end{equation}
geçerlidir.
