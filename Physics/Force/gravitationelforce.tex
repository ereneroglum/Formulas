    F = G \frac{m_1 m_2}{d^2} \qquad \qquad \parbox{4cm}{\footnotesize$\begin{aligned}

        d\text{: Ağırlık Merkezleri Arasındaki Mesafe (m) } \\
\end{aligned}$}
\end{equation}
