\subsection{Kırılma}

\subsubsection*{Kırıcılık İndisi}
\begin{equation}
     n = \frac{c}{V} \qquad \qquad \parbox{4cm}{\footnotesize$\begin{aligned}
          n\text{: Ortamın Kırıcılık İndisi } \\
          c\text{: Işık Hızı } \\
          v\text{: Işığın Ortamda Hızı } (\frac{m}{s})
\end{aligned}$}
\end{equation}

\subsubsection*{Snell Yasası}
\begin{equation}
    n_1 \sin{\alpha} = n_2 \sin{\beta} \qquad \qquad \parbox{4cm}{\footnotesize$\begin{aligned}
          n_1\text{, }n_2\text{: Ortamın Kırıcılık İndisi }
\end{aligned}$}
\end{equation}
