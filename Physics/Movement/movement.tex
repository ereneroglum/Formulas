\subsubsection*{Konum - Hız - İvme İlişkisi}
\begin{equation}
    \frac{d^2 \vec{x}}{dt^2} = \frac{d\vec{v}}{dt} = \vec{a} \qquad \qquad \parbox{4cm}{\footnotesize$\begin{aligned}
        \vec{x}\text{: Konum (m)} \\
        \vec{v}\text{: Hız } (\frac{m}{s}) \\
        \vec{a}\text{: İvme } (\frac{m}{s^2}) \\    
        t\text{: Zaman (s)}
\end{aligned}$}
\end{equation}
