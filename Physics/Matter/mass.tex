\subsection{Kütle}

\subsubsection*{Kütle Özkütle Hacim İlişkisi}

\begin{equation}
    m = dV \qquad \qquad \parbox{4cm}{\footnotesize$
    \begin{aligned} 
        m\text{: Kütle } (kg) \\
        d\text{: Özkütle } (\frac{kg}{m^3} = \frac{g}{dm^3} = \frac{mg}{cm^3}) \\
        V\text{: Hacim } (m^3)
    \end{aligned}$}
\end{equation}
